/** * auth.js - Authentication handlers */ const Auth = { login(email, pin) { return new Promise((resolve, reject) => { google.script.run .withSuccessHandler(resolve) .withFailureHandler(reject) .handleLogin({ email, pin }); }); }, logout() { return new Promise((resolve, reject) => { google.script.run .withSuccessHandler(resolve) .withFailureHandler(reject) .handleLogout(); }); } };
